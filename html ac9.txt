          carousel.addEventListener('mouseup', () => {

          carousel.addEventListener('mousemove', (e) => {
            if (!isDown) return;

      // 使用提供的 shuffleImageOrder 函数
 function shuffleImageOrder() {
        const brandImages = document.querySelectorAll('.brandImage');
        const parentInfo = [];
        // Store the parent node and next sibling for each brand image for reinsertion
        brandImages.forEach(node => {
          parentInfo.push({
            parent: node.parentNode,
            nextSibling: node.nextSibling
          });
        });
        // Shuffle function (Fisher-Yates/Durstenfeld)
        for (let i = brandImages.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          [brandImages[i], brandImages[j]] = [brandImages[j], brandImages[i]];
          [parentInfo[i], parentInfo[j]] = [parentInfo[j], parentInfo[i]];
        }
        // Reinsert shuffled brand images in their original containers
        brandImages.forEach((node, index) => {
          const {
            parent,
            nextSibling
          } = parentInfo[index];
          if (nextSibling) {
            parent.insertBefore(node, nextSibling);
          } else {
            parent.appendChild(node);
          }
        });
      }
      shuffleImageOrder();
